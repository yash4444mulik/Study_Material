		CONSTRAINT rno_PK PRIMARY KEY(rno),

		CONSTRAINT Gno_FK FOREIGN KEY(Gno) REFERENCES Guest(Gno));

	select roomtype,rno from Room where rno=101

	Q. Find the name of guest who have allocated room more than three days?
	select gname,no_of_days from Guest where no_of_days>3

	select rno,roomtype,rate from Room where Rate=(select max(rate) from Room)

	Q. List room wise guest name?
