Q.. Write a program using explicit cursor to handle NO_DATA_FOUND and TOO_MANY_ROWS exceptions.
==> declare
	cursor v_cursor IS SELECT empname,salary from employee where empno=101;
	v_empname varchar2(90);
	v_salary number;
	v_count number:=0;
	begin
	open v_cursor;

	loop

	fetch v_cursor into v_empname,v_salary;

	exit when v_cursor%NOTFOUND;

	v_count:=v_count+1;
	dbms_output.put_line(v_empname||v_salary);

	end loop;

	close v_cursor;

	if v_count=0 then 
	raise NO_DATA_FOUND;
	elsif v_count>1 then
	raise TOO_MANY_ROWS;
	end if;

	EXCEPTION
	when NO_DATA_FOUND then
	dbms_output.put_line('NO_DATA_FOUND');
	when TOO_MANY_ROWS then
	dbms_output.put_line('TOO_MANY_VALUES');
	end;

Q.. write explicit procedure using cursor which take empno as input and show salary as output
	use: 1 cursor,%TYPE,IF-ELSE if salary>50000 high else return low,handle no_data_found,too_many_rows,others.
	print employee name inside procedure,ensure that cursor is open then close it?
==> create or replace procedure p_salary_op(p_empno IN NUMBER,p_salary OUT NUMBER,p_status OUT VARCHAR2)
	IS
	cursor v_cursor IS SELECT salary,empname from employee where empno=p_empno;
	v_salary employee.salary%TYPE;
	v_empname employee.empname%TYPE;
	v_count number:=0;
	BEGIN

	OPEN v_cursor;

	LOOP
	fetch v_cursor into v_salary,v_empname;
	exit when v_cursor%NOTFOUND;

	v_count:=v_count+1;

	p_salary:=v_salary;
	dbms_output.put_line('employee name is: '||v_empname);

	if v_salary>50000 then
	p_status:='salary is high';
	else
	p_status:='salary is low';
	end if;

	END LOOP;

	close v_cursor;

	if v_count=0 then 
	raise NO_DATA_FOUND;
	elsif v_count>1 then
	raise TOO_MANY_ROWS;
	end if;

	EXCEPTION
	when NO_DATA_FOUND then 
	dbms_output.put_line('no data found for mentioned empno');
	p_salary:=null;
	p_status := 'No data found for given empno';
	when TOO_MANY_ROWS then
	dbms_output.put_line('too many employees belongs to this empno');
	p_status := 'Multiple employees found';
	p_salary:=null;
	when OTHERS then 
	p_salary:=null;
	p_status := 'Error: ' || SQLERRM;

	if v_cursor%ISOPEN then
	close v_cursor;
	end if;
	END;
	/


	declare
	o_salary number;
	o_status varchar2(90);
	begin
	p_salary_op(101,o_salary,o_status);
	dbms_output.put_line('salary: '||o_salary);
	dbms_output.put_line('status: '||o_status);

	v_emp emp_rec;
	BEGIN
	open emp_cursor;
	LOOP
	fetch emp_cursor bulk collect into v_emp limit 5;
	exit when v_emp.count=0;
	dbms_output.put_line('----Batch----');
	for i IN 1..v_emp.count loop
	dbms_output.put_line(v_emp(i).empno);
	end loop;
	END LOOP;
