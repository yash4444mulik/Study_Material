Q.. Write a program using explicit cursor to handle NO_DATA_FOUND and TOO_MANY_ROWS exceptions.
==> declare
	cursor v_cursor IS SELECT empname,salary from employee where empno=101;
	v_empname varchar2(90);
	v_salary number;
	v_count number:=0;
	begin
	open v_cursor;

	loop

	fetch v_cursor into v_empname,v_salary;

	exit when v_cursor%NOTFOUND;

	v_count:=v_count+1;
	dbms_output.put_line(v_empname||v_salary);

	end loop;

	close v_cursor;

	if v_count=0 then 
	raise NO_DATA_FOUND;
	elsif v_count>1 then
	raise TOO_MANY_ROWS;
	end if;

	EXCEPTION
	when NO_DATA_FOUND then
	dbms_output.put_line('NO_DATA_FOUND');
	when TOO_MANY_ROWS then
	dbms_output.put_line('TOO_MANY_VALUES');
	end;

Q.. write explicit procedure using cursor which take empno as input and show salary as output
	use: 1 cursor,%TYPE,IF-ELSE if salary>50000 high else return low,handle no_data_found,too_many_rows,others.
	print employee name inside procedure,ensure that cursor is open then close it?
==> create or replace procedure p_salary_op(p_empno IN NUMBER,p_salary OUT NUMBER,p_status OUT VARCHAR2)
	IS
	cursor v_cursor IS SELECT salary,empname from employee where empno=p_empno;
	v_salary employee.salary%TYPE;
	v_empname employee.empname%TYPE;
	v_count number:=0;
	BEGIN

	OPEN v_cursor;

	LOOP
	fetch v_cursor into v_salary,v_empname;
	exit when v_cursor%NOTFOUND;

	v_count:=v_count+1;

	p_salary:=v_salary;
	dbms_output.put_line('employee name is: '||v_empname);

	if v_salary>50000 then
	p_status:='salary is high';
	else
	p_status:='salary is low';
	end if;

	END LOOP;

	close v_cursor;

	if v_count=0 then 
	raise NO_DATA_FOUND;
	elsif v_count>1 then
	raise TOO_MANY_ROWS;
	end if;

	EXCEPTION
	when NO_DATA_FOUND then 
	dbms_output.put_line('no data found for mentioned empno');
	p_salary:=null;
	p_status := 'No data found for given empno';
	when TOO_MANY_ROWS then
	dbms_output.put_line('too many employees belongs to this empno');
	p_status := 'Multiple employees found';
	p_salary:=null;
	when OTHERS then 
	p_salary:=null;
	p_status := 'Error: ' || SQLERRM;

	if v_cursor%ISOPEN then
	close v_cursor;
	end if;
	END;

==> CREATE OR REPLACE FUNCTION f_palindrome(f_num NUMBER)return varchar2
	IS
	v_temp number;
	v_reverse number:=0;
	v_rem number;
	BEGIN
	v_temp:=f_num;
	WHILE v_temp>0 LOOP
	v_rem:=mod(v_temp,10);
	v_reverse:=v_reverse*10+v_rem;
	v_temp:=trunc(v_temp/10);
	END LOOP;
	IF v_reverse=f_num THEN
	RETURN 'palindrom';
	ELSE
	RETURN 'not palindrom';
	END IF;
	END;

	IS
	f_len NUMBER;
	result VARCHAR2(90);
	BEGIN
	f_len:=length(f_string);
	FOR i IN REVERSE 1..f_len LOOP
	result:=concat(result,substr(f_string,i,1));

49. Write a procedure to transfer an amount between two accounts.
==>	create or replace procedure amount_transfer(first_acc IN NUMBER,second_acc IN NUMBER,p_amount IN NUMBER)
	IS 
	v_balance NUMBER;
	BEGIN
	select balance into v_balance from accounts where acc_no=first_acc;
	IF v_balance<p_amount THEN 
	RAISE_APPLICATION_ERROR(-20001,'Insufficient balance');
	END IF;
	UPDATE accounts set balance=balance-p_amount where acc_no=first_acc;
	UPDATE accounts set balance=balance+p_amount where acc_no=second_acc;
	COMMIT;
	dbms_output.put_line('Amount transferred successfully');
	EXCEPTION
	WHEN NO_DATA_FOUND THEN 
		ROLLBACK;
		dbms_output.put_line('Account not found');
	WHEN OTHERS THEN 
		ROLLBACK;
		dbms_output.put_line(SQLERRM);
	END;
