Q.. What is predefined exception in Oracle?
==> Exception defined by oracle engine and have error code and error name is nothing but predefined exception
	- These exception handled in exception block without declaring them 
	1. NO_DATA_FOUND => select statement return no row 
	2. TOO_MANY_ROWS => select statement return more than one rows
	3. ZERO_DIVIDE => division by zero
	4. DUP_VAL_ON_INDEX => duplicate value in unique index 
	5. INVALID_NUMBER => Invalid number conversion 
	6. VALUE_ERROR => arithmatic,conversion or size error
	
Q.. If i insert duplicate value in block then which type of error occur?
==> If we are inserting duplicate value in primary key and unique key column then oracle 
	raise predefined exception 
	Exception ==> DUP_VAL_ON_INDEX
		begin
		insert into department values(10,'mech','pune');
		exception
		when dup_val_on_index then
		dbms_output.put_line('you are inserting duplicate value in table');
		end;
	
Q.. If error occured then how to handle it?
==> If error occured then we can handle it in exception section 
	- Predefined Exception - we can handle using when exception_name then statement.
		begin
		insert into department values(10,'mech','pune');
		exception
		when dup_val_on_index then
		dbms_output.put_line('you are inserting duplicate value in table');
		end;
	- Non Predefined exception - we can handle using PRAGMA EXCEPTION_INIT
		declare
		fk_exception exception;
		pragma exception_init(fk_exception,-02292);
		begin
		delete from department where deptno=30;
		exception
		WHEN fk_exception then
		dbms_output.put_line('can not delete child recored');
		end;
		
	- User defined exception
		DECLARE
		   e_invalid_salary EXCEPTION;
		   v_salary NUMBER := -1000;

		BEGIN

		   IF v_salary < 0 THEN
			  RAISE e_invalid_salary;
		   END IF;

		EXCEPTION
		   WHEN e_invalid_salary THEN
			  DBMS_OUTPUT.PUT_LINE('Salary cannot be negative');
		END;

Q.. Find out the monthly transactions count?
	select to_char(txn_date,'Month')as month,count(*)as total_transactions from transactions 
	group by to_char(txn_date,'Month')
Q.. Find out the monthly transactions amount?	
	select to_char(txn_date,'Month')as month,sum(amount)as monthly_trans_amount from transactions 
	group by to_char(txn_date,'Month')
Q.. Find mothly transactions running total?
	select 
	month,
	total_amount,
	sum(total_amount) over(order by month_num)as running_totoal
	from(
	select
	to_char(txn_date,'Month')as month,
	sum(amount)as total_amount,
	extract(month from txn_date)as month_num from transactions 
	group by to_char(txn_date,'Month'),extract(month from txn_date)
	)order by month_num
